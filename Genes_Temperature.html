Load the data

therm = data.frame(read.csv('../data-raw/Thermal_pics/allthermal.csv',stringsAsFactors = FALSE))
therm$X=NULL
therm$tomerge=paste(therm$Pot,therm$Tray,sep="_")

gen = data.frame(read.csv('../data-raw/Genes_Lines.csv', sep = ';'  ,stringsAsFactors = FALSE))
gen$Tray = gen$tray
gen$tray = NULL

Translate trays

gen$Tray = gsub("D1", "8.W.A.3.a", gen$Tray)
gen$Tray = gsub("W1", "8.W.A.3.b", gen$Tray)
gen$Tray = gsub("D2", "8.W.A.3.c", gen$Tray)
gen$Tray = gsub("W2", "8.W.B.3.a", gen$Tray)
gen$Tray = gsub("D3", "8.W.B.3.b", gen$Tray)
gen$Tray = gsub("W3", "8.W.B.3.c", gen$Tray)
gen$Tray = gsub("D4", "8.W.C.3.a", gen$Tray)
gen$Tray = gsub("W4", "8.W.C.3.b", gen$Tray)
gen$Tray = gsub("D5", "8.W.C.3.c", gen$Tray)
gen$Tray = gsub("W5", "8.W.D.3.a", gen$Tray)
gen$Tray = gsub("D6", "8.W.D.3.b", gen$Tray)
gen$Tray = gsub("W6", "8.W.D.3.c", gen$Tray)

Merge files

gen$tomerge=paste(gen$pos,c(gen$Tray),sep="_")
genx=merge(x=gen,y=therm, by=c('tomerge'))
stats=subset(genx, id != 'IGN' & mean !='NaN')
stats$Tray.y=NULL
stats$Pot=stats$pos
stats$pos=NULL

mediagen = stats %>%
  group_by(id,watering,Date) %>%
  summarize(meanMeann=mean(mean),sdMean=sd(mean))

Filter out accessions with less than two replicates

mediagenfiltacc = mediagen %>%
  filter(!id %in% c("9944", "SALK_045606", "SALK_060946", "SALK_107255", "SALK_137176"))

Filter per date

mediagenfilt=dplyr::filter(mediagenfiltacc, Date=="08-22")

Plot data - remember to change yintercept according to date (Col-0)

genmediaplotw=ggplot(mediagenfilt, aes(y=meanMeann, x=id, group=Date, color=meanMeann)) +
  geom_point(data=subset(mediagenfilt,watering=="W")) + theme_bw() +
  theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
  scale_colour_gradient(low = "blue",high = "orange", limits=c(18, 26)) + ylab("meanMeann Mean") +
  ylim(18,26) + geom_hline(yintercept = 21.53186, color="dark grey") +
  geom_errorbar(data=subset(mediagenfilt,watering=="W"), aes(ymin=meanMeann-sdMean, ymax=meanMeann+sdMean))

genmediaplotd=ggplot(mediagenfilt, aes(y=meanMeann, x=id, group=Date, color=meanMeann)) +
  geom_point(data=subset(mediagenfilt,watering=="D")) + theme_bw() +
  theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
  scale_colour_gradient(low = "blue",high = "orange", limits=c(18, 26)) + ylab("meanMeann Mean") +
  ylim(18,26) + geom_hline(yintercept = 21.74843, color="dark grey") +

genmediafin=plot_grid(genmediaplotw,genmediaplotd,labels=c("W","D"),ncol = 2)
